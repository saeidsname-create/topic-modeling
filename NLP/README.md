## فایل اصلی برنامه (detect.py)

```python
text = input("متن را وارد کنید: ")


latin = arabic = persian = cyrillic = other = 0

for ch in text:
    code = ord(ch)
* در این حلقه کاراکتر ها رو دریافت کرده و هرکدام را به ascii code یا unicode تغییر میدهد تا بتواند حساب کتاب کند
    
    if (65 <= code <= 90) or (97 <= code <= 122):
        latin += 1
* در این مرحله ascii code ها رو مقایسه میکند تا بفهمد این حروف انگلیسی حرف های بزرگ دارد یا کوچک تا همه را پوشش دهد
    
    elif 0x0600 <= code <= 0x06FF:
        persian += 1
* در این مرحله unicode ها رو مقایسه میکند تا بفهمد حروف فارسی وجود دارد یا خیر
    
    elif 0x0400 <= code <= 0x04FF:
        cyrillic += 1
* در این مرحله هم باز unicode ها رو چک میکند تا بفهمد حروف عربی در آن وجود دارد یا خیر
    
    else:
        other += 1
* اگر حروف دیگه ای غیر از این ها وجود داشت به other اضافه میکنه تا بگه زبون های دیگه ای هم توی این کد وجود دارد

print("\nنتیجه:")
if latin > 0: print("حروف لاتین وجود دارد.")
if persian > 0: print("حروف فارسی/عربی وجود دارد.")
if cyrillic > 0: print("حروف سیریلیک وجود دارد.")
if other > 0: print("نماد/زبان‌های دیگر هم دیده شد.")
